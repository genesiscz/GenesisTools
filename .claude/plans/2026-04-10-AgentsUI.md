# Agents UI — Unified Formatter & Component Architecture

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Extract an agent-agnostic formatting and component layer that unifies rendering across `tools claude tail`, `tools ask`, and `claude-history-dashboard` — and is ready for Codex/other agents.

**Architecture:** A three-layer stack: (1) `AgentMessage` — normalized message model that any agent session (Claude, Codex, generic) can map into; (2) `Formatter` — transforms `AgentMessage[]` into `FormattedBlock[]` (structured, renderer-agnostic); (3) `Renderer` — terminal (`TerminalRenderer`) or web (`WebRenderer`) that converts blocks to output. React components consume `FormattedBlock[]` directly.

**Tech Stack:** TypeScript, picocolors (terminal), React + Tailwind + shadcn/ui (web), existing `src/utils/claude/types.ts` as Claude adapter source.

```ts
    | { type: "text"; text: string }

    role: "user" | "assistant" | "system" | "metadata";
    blocks: AgentContentBlock[];
    timestamp?: Date;
    model?: string;
    usage?: AgentUsage;

    inputTokens: number;

    project?: string;
```

```ts
// src/utils/agents/formatters/block-parser.ts
import type { AgentMessage } from "../types";
import type { FormattedBlock, FormatOptions } from "./types";
import { formatToolSignature, formatToolDiff, formatToolResult } from "./tool-formatter";

/** Convert an AgentMessage into FormattedBlocks for rendering. */
export function messageToBlocks(msg: AgentMessage, options: FormatOptions): FormattedBlock[] {
    const blocks: FormattedBlock[] = [];

    if (options.showRoleHeaders) {
        blocks.push({
            type: "role-header",
            content: msg.role === "user" ? "User" : "Assistant",
            meta: { role: msg.role, timestamp: msg.timestamp, model: msg.model },
        });
    }

    for (const block of msg.blocks) {
        switch (block.type) {
            case "text":
                blocks.push({ type: "text", content: block.text, meta: { role: msg.role } });
                break;
            case "thinking":
                if (options.showThinking) {
                    blocks.push({ type: "thinking", content: block.text });
                }
                break;
            case "tool_call": {
                const sig = formatToolSignature(block.name, block.input, {
                    primaryMaxChars: options.toolInputMaxChars ?? 500,
                    detailLevel: options.toolDetailLevel ?? "signature",
                });
                blocks.push({ type: "tool-signature", content: sig, meta: { toolName: block.name, filePath: block.input.file_path as string } });
                // Diff lines for Edit/Write at full detail
                const diffLines = formatToolDiff(block.name, block.input, options.toolInputMaxChars ?? 500);
                if (diffLines) {
                    blocks.push({ type: "tool-diff", content: diffLines[0], lines: diffLines, meta: { toolName: block.name } });
                }
                break;
            }
            case "tool_result": {
                const result = formatToolResult(block.content, options.toolOutputMaxChars ?? 500, { isError: block.isError });
                blocks.push({ type: "tool-result", content: result, meta: { isError: block.isError } });
                break;
            }
            case "image":
                blocks.push({ type: "image", content: `[image: ${block.mediaType}]`, meta: {} });
                break;
        }
    }

    return blocks;
}

/** Convert a full conversation to blocks. */
export function conversationToBlocks(messages: AgentMessage[], options: FormatOptions): FormattedBlock[] {
    return messages.flatMap((msg) => messageToBlocks(msg, options));
}
```

```ts
// src/utils/agents/renderers/TerminalRenderer.ts
import pc from "picocolors";
import type { FormattedBlock } from "../formatters/types";

export interface TerminalRenderOptions {
    colors: boolean;
    indent?: string;
    /** Compact mode — fewer separators, shorter output. */
    compact?: boolean;
}

export class TerminalRenderer {
    constructor(private options: TerminalRenderOptions) {}

    /** Render a single block to terminal string(s). */
    renderBlock(block: FormattedBlock): string[] {
        const c = this.options.colors;
        const indent = this.options.indent ?? "";

        switch (block.type) {
            case "role-header": {
                const label = block.meta?.role === "user" ? "You" : "Claude";
                const icon = block.meta?.role === "user" ? "›" : "⏺";
                return [c ? `${indent}${pc.dim(icon)} ${pc.bold(label)}` : `${indent}${icon} ${label}`];
            }
            case "text":
                return block.content.split("\n").map(line =>
                    c && block.meta?.role === "user" ? `${indent}${pc.green(line)}` : `${indent}${line}`
                );
            case "thinking":
                return block.content.split("\n").map(line =>
                    c ? `${indent}${pc.dim(pc.italic(line))}` : `${indent}${line}`
                );
            case "tool-signature": {
                const colored = c ? this.colorizeSignature(block) : `⏺ ${block.content}`;
                return [`${indent}${colored}`];
            }
            case "tool-diff":
                return (block.lines ?? []).map(line => {
                    if (c) {
                        if (line.startsWith("+")) return `${indent}  ${pc.green(line)}`;
                        if (line.startsWith("-")) return `${indent}  ${pc.red(line)}`;
                    }
                    return `${indent}  ${line}`;
                });
            case "tool-result": {
                const text = block.content;
                if (block.meta?.isError) {
                    return text.split("\n").map(l => c ? `${indent}  ${pc.red(l)}` : `${indent}  ${l}`);
                }
                return text.split("\n").map(l => c ? `${indent}  ${pc.dim(l)}` : `${indent}  ${l}`);
            }
            case "separator":
                return [""];
            default:
                return block.content ? [`${indent}${block.content}`] : [];
        }
    }

    /** Render multiple blocks. */
    render(blocks: FormattedBlock[]): string[] {
        return blocks.flatMap(b => this.renderBlock(b));
    }

    private colorizeSignature(block: FormattedBlock): string {
        const name = block.meta?.toolName ?? "";
        const toolColor = { Read: pc.cyan, Grep: pc.cyan, Glob: pc.cyan, Edit: pc.yellow, Write: pc.yellow, Bash: pc.magenta, Agent: pc.green }[name] ?? pc.dim;
        const parenIdx = block.content.indexOf("(");
        if (parenIdx === -1) return `${pc.blue("⏺")} ${toolColor(block.content)}`;
        return `${pc.blue("⏺")} ${toolColor(block.content.slice(0, parenIdx))}${pc.dim(block.content.slice(parenIdx))}`;
    }
}
```

```ts
// src/utils/agents/renderers/WebRenderer.ts
import type { FormattedBlock } from "../formatters/types";

/** Renders FormattedBlocks as HTML strings with CSS class names for styling. */
export class WebRenderer {
    renderBlock(block: FormattedBlock): string {
        switch (block.type) {
            case "role-header": {
                const cls = block.meta?.role === "user" ? "msg-user" : "msg-assistant";
                return `<div class="role-header ${cls}">${this.escape(block.content)}</div>`;
            }
            case "text":
                return `<div class="msg-text">${this.escape(block.content)}</div>`;
            case "thinking":
                return `<details class="thinking-block"><summary>Thinking</summary><pre>${this.escape(block.content)}</pre></details>`;
            case "tool-signature":
                return `<div class="tool-signature" data-tool="${this.escape(block.meta?.toolName ?? "")}">${this.escape(block.content)}</div>`;
            case "tool-diff": {
                const lines = (block.lines ?? []).map(l => {
                    const cls = l.startsWith("+") ? "diff-add" : l.startsWith("-") ? "diff-remove" : "diff-context";
                    return `<span class="${cls}">${this.escape(l)}</span>`;
                }).join("\n");
                return `<pre class="tool-diff">${lines}</pre>`;
            }
            case "tool-result": {
                const cls = block.meta?.isError ? "tool-result error" : "tool-result";
                return `<pre class="${cls}">${this.escape(block.content)}</pre>`;
            }
            default:
                return block.content ? `<div>${this.escape(block.content)}</div>` : "";
        }
    }

    render(blocks: FormattedBlock[]): string {
        return blocks.map(b => this.renderBlock(b)).join("\n");
    }

    private escape(s: string): string {
        return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");
    }
}
```

## Future: Adding a new agent provider

To add Codex (or any other agent):

1. Create `src/utils/agents/adapters/codex.ts` — map Codex response format → `AgentMessage[]`
2. That's it. Formatters, renderers, and React components work unchanged.

The `AgentMessage` model is the stable interface. Wire formats come and go — the rendering layer doesn't need to know.
